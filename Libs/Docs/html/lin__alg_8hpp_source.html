94 T max = fabs(M[current_row][current_row]), maxI = current_row;

116template <class T>
117void gaussElim(T **M, T v[], const int& nRows) {
118 // Reduce M to upper triangular form
119 for (int k = 0; k < nRows - 1; k++) {
120 partialPivoting(M, v, nRows, k);
121 for (int i = k + 1; i < nRows; i++) {
122 T g = M[i][k] / M[k][k];
123 for (int j = k + 1; j < nRows; j++) M[i][j] -= g * M[k][j];
124 M[i][k] = 0.0;
125 v[i] -= g * v[k];
126 }
127 }
128}

141void solveLinSystem(T **M, T v[], T x[], const int& nEqs) {
142 // Reduce the system to upper triangular form
143 gaussElim(M, v, nEqs);
144
145 // Solve the system

147 T temp = v[i];

165template <class T>
166void tridiagonalSolver(T a[], T b[], T c[], T r[], T x[], const int& nEq) {
167 const int maxSize = 64;
168 if (nEq > maxSize) throw std::invalid_argument("Number of equations must not be greater than " + std::to_string(maxSize) + ".");
169 if (!isnan(a[0])) throw std::invalid_argument("The first element of a must be nan(\"\").");
170 if (!isnan(c[nEq - 1])) throw std::invalid_argument("The last element of c must be nan(\"\").");
171 T h[maxSize], p[maxSize];
172
173 h[0] = c[0] / b[0];
174 p[0] = r[0] / b[0];
175
176 // Gaussian elimination
177 for (int i = 1; i < nEq; i++) {
178 T den = 1.0 / (b[i] - a[i] * h[i - 1]);
179 h[i] = c[i] * den;
180 p[i] = (r[i] - a[i] * p[i - 1]) * den;
181 }
182
183 // Backsubsitution
184 x[nEq - 1] = p[nEq - 1];
185 for (int i = nEq - 2; i >= 0; i--) {
186 x[i] = p[i] - h[i] * x[i + 1];
187 }
188}
